import tkinter as tk

window = tk.Tk()
window.title("Temp Converter")
window.geometry("500x300")
window.resizable(width=True,height=False)

def fahrenheit_to_celcius():
    fahrenheit = ent_temperature.get()
    celsius = (5/9) * (float(fahrenheit) - 32 )
    lbl_result["text"] = f"{round(celsius, 2)} \N{DEGREE CELSIUS}"

frm_entry = tk.Frame(master=window)
ent_temperature = tk.Entry(master=frm_entry, width=10)
lbl_temp = tk.Label(master=frm_entry, text="\N{DEGREE FAHRENHEIT} INPUT ")

ent_temperature.grid(row=0, column=0, sticky="e")
lbl_temp.grid(row=0, column=1, sticky="w")

btn_convert = tk.Button(
    master=window, text="Convert temp to:",
    command=fahrenheit_to_celcius
)

lbl_result = tk.Label(master=window, text="\N{DEGREE CELSIUS}  RESULT")


frm_entry.grid(row=0, column=0, padx=10)
btn_convert.grid(row=0, column=1, pady=10)
lbl_result.grid(row=0, column=2, padx=10)


window.mainloop()
